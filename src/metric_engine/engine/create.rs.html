60    pub async fn create_regions(
61        &self,
62        mut requests: Vec<(RegionId, RegionCreateRequest)>,
63        extension_return_value: &mut HashMap<String, Vec<u8>>,
64    ) -> Result<AffectedRows> {
65        if requests.is_empty() {
66            return Ok(0);
67        }
68
69        for (_, request) in requests.iter() {
70            Self::verify_region_create_request(request)?;
71        }
72
73        let first_request = &requests.first().unwrap().1;
74        if first_request.is_physical_table() {
75            ensure!(
76                requests.len() == 1,
77                UnexpectedRequestSnafu {
78                    reason: "Physical table must be created with single request".to_string(),
79                }
80            );
81            let (region_id, request) = requests.pop().unwrap();
82            self.create_physical_region(region_id, request).await?;
83
84            return Ok(0);
85        } else if first_request
86            .options
87            .contains_key(LOGICAL_TABLE_METADATA_KEY)
88        {
89            if requests.len() == 1 {
90                let request = &requests.first().unwrap().1;
91                let physical_region_id = parse_physical_region_id(request)?;
92                let mut manifest_infos = Vec::with_capacity(1);
93                self.create_logical_regions(physical_region_id, requests, extension_return_value)
94                    .await?;
95                append_manifest_info(&self.mito, physical_region_id, &mut manifest_infos);
96                encode_manifest_info_to_extensions(&manifest_infos, extension_return_value)?;
97            } else {
98                let grouped_requests =
99                    group_create_logical_region_requests_by_physical_region_id(requests)?;
100                let mut manifest_infos = Vec::with_capacity(grouped_requests.len());
101                for (physical_region_id, requests) in grouped_requests {
102                    self.create_logical_regions(
103                        physical_region_id,
104                        requests,
105                        extension_return_value,
106                    )
107                    .await?;
108                    append_manifest_info(&self.mito, physical_region_id, &mut manifest_infos);
109                }
110                encode_manifest_info_to_extensions(&manifest_infos, extension_return_value)?;
111            }
112        } else {
113            return MissingRegionOptionSnafu {}.fail();
114        }
115
116        Ok(0)
117    }

119    /// Initialize a physical metric region at given region id.
120    async fn create_physical_region(
121        &self,
122        region_id: RegionId,
123        request: RegionCreateRequest,
124    ) -> Result<()> {
125        let physical_region_options = PhysicalRegionOptions::try_from(&request.options)?;
126        let (data_region_id, metadata_region_id) = Self::transform_region_id(region_id);
127
128        // create metadata region
129        let create_metadata_region_request = self.create_request_for_metadata_region(&request);
130        self.mito
131            .handle_request(
132                metadata_region_id,
133                RegionRequest::Create(create_metadata_region_request),
134            )
135            .await
136            .with_context(|_| CreateMitoRegionSnafu {
137                region_type: METADATA_REGION_SUBDIR,
138            })?;
139
140        // create data region
141        let create_data_region_request = self.create_request_for_data_region(&request);
142        let physical_columns = create_data_region_request
143            .column_metadatas
144            .iter()
145            .map(|metadata| (metadata.column_schema.name.clone(), metadata.column_id))
146            .collect::<HashMap<_, _>>();
147        let time_index_unit = create_data_region_request
148            .column_metadatas
149            .iter()
150            .find_map(|metadata| {
151                if metadata.semantic_type == SemanticType::Timestamp {
152                    metadata
153                        .column_schema
154                        .data_type
155                        .as_timestamp()
156                        .map(|data_type| data_type.unit())
157                } else {
158                    None
159                }
160            })
161            .context(UnexpectedRequestSnafu {
162                reason: "No time index column found",
163            })?;
164        self.mito
165            .handle_request(
166                data_region_id,
167                RegionRequest::Create(create_data_region_request),
168            )
169            .await
170            .with_context(|_| CreateMitoRegionSnafu {
171                region_type: DATA_REGION_SUBDIR,
172            })?;
173        let primary_key_encoding = self.mito.get_primary_key_encoding(data_region_id).context(
174            PhysicalRegionNotFoundSnafu {
175                region_id: data_region_id,
176            },
177        )?;
178
179        info!("Created physical metric region {region_id}, primary key encoding={primary_key_encoding}, physical_region_options={physical_region_options:?}");
180        PHYSICAL_REGION_COUNT.inc();
181
182        // remember this table
183        self.state.write().unwrap().add_physical_region(
184            data_region_id,
185            physical_columns,
186            primary_key_encoding,
187            physical_region_options,
188            time_index_unit,
189        );
190
191        Ok(())
192    }

194    /// Create multiple logical regions on the same physical region.
195    async fn create_logical_regions(
196        &self,
197        physical_region_id: RegionId,
198        requests: Vec<(RegionId, RegionCreateRequest)>,
199        extension_return_value: &mut HashMap<String, Vec<u8>>,
200    ) -> Result<()> {
201        let data_region_id = utils::to_data_region_id(physical_region_id);
202
203        let unit = self
204            .state
205            .read()
206            .unwrap()
207            .physical_region_time_index_unit(physical_region_id)
208            .context(PhysicalRegionNotFoundSnafu {
209                region_id: data_region_id,
210            })?;
211        // Checks the time index unit of each request.
212        for (_, request) in &requests {
213            // Safety: verify_region_create_request() ensures that the request is valid.
214            let time_index_column = request
215                .column_metadatas
216                .iter()
217                .find(|col| col.semantic_type == SemanticType::Timestamp)
218                .unwrap();
219            let request_unit = time_index_column
220                .column_schema
221                .data_type
222                .as_timestamp()
223                .unwrap()
224                .unit();
225            ensure!(
226                request_unit == unit,
227                UnexpectedRequestSnafu {
228                    reason: format!(
229                        "Metric has differenttime unit ({:?}) than the physical region ({:?})",
230                        request_unit, unit
231                    ),
232                }
233            );
234        }
235
236        // Filters out the requests that the logical region already exists
237        let requests = {
238            let state = self.state.read().unwrap();
239            let mut skipped = Vec::with_capacity(requests.len());
240            let mut kept_requests = Vec::with_capacity(requests.len());
241
242            for (region_id, request) in requests {
243                if state.is_logical_region_exist(region_id) {
244                    skipped.push(region_id);
245                } else {
246                    kept_requests.push((region_id, request));
247                }
248            }
249
250            // log skipped regions
251            if !skipped.is_empty() {
252                info!(
253                    "Skipped creating logical regions {skipped:?} because they already exist",
254                    skipped = skipped
255                );
256            }
257            kept_requests
258        };
259
260        // Finds new columns to add to physical region
261        let mut new_column_names = HashSet::new();
262        let mut new_columns = Vec::new();
263
264        let index_option = {
265            let state = &self.state.read().unwrap();
266            let region_state = state
267                .physical_region_states()
268                .get(&data_region_id)
269                .with_context(|| PhysicalRegionNotFoundSnafu {
270                    region_id: data_region_id,
271                })?;
272            let physical_columns = region_state.physical_columns();
273
274            extract_new_columns(
275                &requests,
276                physical_columns,
277                &mut new_column_names,
278                &mut new_columns,
279            )?;
280            region_state.options().index
281        };
282
283        // TODO(weny): we dont need to pass a mutable new_columns here.
284        self.data_region
285            .add_columns(data_region_id, new_columns, index_option)
286            .await?;
287
288        let physical_columns = self.data_region.physical_columns(data_region_id).await?;
289        let physical_schema_map = physical_columns
290            .iter()
291            .map(|metadata| (metadata.column_schema.name.as_str(), metadata))
292            .collect::<HashMap<_, _>>();
293        let logical_regions = requests
294            .iter()
295            .map(|(region_id, _)| (*region_id))
296            .collect::<Vec<_>>();
297        let logical_region_columns = requests.iter().map(|(region_id, request)| {
298            (
299                *region_id,
300                request
301                    .column_metadatas
302                    .iter()
303                    .map(|metadata| {
304                        // Safety: previous steps ensure the physical region exist
305                        let column_metadata = *physical_schema_map
306                            .get(metadata.column_schema.name.as_str())
307                            .unwrap();
308                        (metadata.column_schema.name.as_str(), column_metadata)
309                    })
310                    .collect::<HashMap<_, _>>(),
311            )
312        });
313
314        let new_add_columns = new_column_names.iter().map(|name| {
315            // Safety: previous steps ensure the physical region exist
316            let column_metadata = *physical_schema_map.get(name).unwrap();
317            (name.to_string(), column_metadata.column_id)
318        });
319
320        extension_return_value.insert(
321            ALTER_PHYSICAL_EXTENSION_KEY.to_string(),
322            ColumnMetadata::encode_list(&physical_columns).context(SerializeColumnMetadataSnafu)?,
323        );
324
325        // Writes logical regions metadata to metadata region
326        self.metadata_region
327            .add_logical_regions(physical_region_id, true, logical_region_columns)
328            .await?;
329
330        {
331            let mut state = self.state.write().unwrap();
332            state.add_physical_columns(data_region_id, new_add_columns);
333            state.add_logical_regions(physical_region_id, logical_regions.clone());
334        }
335        for logical_region_id in logical_regions {
336            self.metadata_region
337                .open_logical_region(logical_region_id)
338                .await;
339        }
340
341        Ok(())
342    }

344    /// Check if
345    /// - internal columns are not occupied
346    /// - required table option is present ([PHYSICAL_TABLE_METADATA_KEY] or
347    ///   [LOGICAL_TABLE_METADATA_KEY])
348    fn verify_region_create_request(request: &RegionCreateRequest) -> Result<()> {
349        request.validate().context(InvalidMetadataSnafu)?;
350
351        let name_to_index = request
352            .column_metadatas
353            .iter()
354            .enumerate()
355            .map(|(idx, metadata)| (metadata.column_schema.name.clone(), idx))
356            .collect::<HashMap<String, usize>>();
357
358        // check if internal columns are not occupied
359        ensure!(
360            !name_to_index.contains_key(DATA_SCHEMA_TABLE_ID_COLUMN_NAME),
361            InternalColumnOccupiedSnafu {
362                column: DATA_SCHEMA_TABLE_ID_COLUMN_NAME,
363            }
364        );
365        ensure!(
366            !name_to_index.contains_key(DATA_SCHEMA_TSID_COLUMN_NAME),
367            InternalColumnOccupiedSnafu {
368                column: DATA_SCHEMA_TSID_COLUMN_NAME,
369            }
370        );
371
372        // check if required table option is present
373        ensure!(
374            request.is_physical_table() || request.options.contains_key(LOGICAL_TABLE_METADATA_KEY),
375            MissingRegionOptionSnafu {}
376        );
377        ensure!(
378            !(request.is_physical_table()
379                && request.options.contains_key(LOGICAL_TABLE_METADATA_KEY)),
380            ConflictRegionOptionSnafu {}
381        );
382
383        // check if only one field column is declared, and all tag columns are string
384        let mut field_col: Option<&ColumnMetadata> = None;
385        for col in &request.column_metadatas {
386            match col.semantic_type {
387                SemanticType::Tag => ensure!(
388                    col.column_schema.data_type == ConcreteDataType::string_datatype(),
389                    ColumnTypeMismatchSnafu {
390                        expect: ConcreteDataType::string_datatype(),
391                        actual: col.column_schema.data_type.clone(),
392                    }
393                ),
394                SemanticType::Field => {
395                    if field_col.is_some() {
396                        MultipleFieldColumnSnafu {
397                            previous: field_col.unwrap().column_schema.name.clone(),
398                            current: col.column_schema.name.clone(),
399                        }
400                        .fail()?;
401                    }
402                    field_col = Some(col)
403                }
404                SemanticType::Timestamp => {}
405            }
406        }
407        let field_col = field_col.context(NoFieldColumnSnafu)?;
408
409        // make sure the field column is float64 type
410        ensure!(
411            field_col.column_schema.data_type == ConcreteDataType::float64_datatype(),
412            ColumnTypeMismatchSnafu {
413                expect: ConcreteDataType::float64_datatype(),
414                actual: field_col.column_schema.data_type.clone(),
415            }
416        );
417
418        Ok(())
419    }

431    /// Build [RegionCreateRequest] for metadata region
432    ///
433    /// This method will append [METADATA_REGION_SUBDIR] to the given `region_dir`.
434    pub fn create_request_for_metadata_region(
435        &self,
436        request: &RegionCreateRequest,
437    ) -> RegionCreateRequest {
438        // ts TIME INDEX DEFAULT 0
439        let timestamp_column_metadata = ColumnMetadata {
440            column_id: METADATA_SCHEMA_TIMESTAMP_COLUMN_INDEX as _,
441            semantic_type: SemanticType::Timestamp,
442            column_schema: ColumnSchema::new(
443                METADATA_SCHEMA_TIMESTAMP_COLUMN_NAME,
444                ConcreteDataType::timestamp_millisecond_datatype(),
445                false,
446            )
447            .with_default_constraint(Some(datatypes::schema::ColumnDefaultConstraint::Value(
448                Value::Timestamp(Timestamp::new_millisecond(0)),
449            )))
450            .unwrap(),
451        };
452        // key STRING PRIMARY KEY
453        let key_column_metadata = ColumnMetadata {
454            column_id: METADATA_SCHEMA_KEY_COLUMN_INDEX as _,
455            semantic_type: SemanticType::Tag,
456            column_schema: ColumnSchema::new(
457                METADATA_SCHEMA_KEY_COLUMN_NAME,
458                ConcreteDataType::string_datatype(),
459                false,
460            ),
461        };
462        // val STRING
463        let value_column_metadata = ColumnMetadata {
464            column_id: METADATA_SCHEMA_VALUE_COLUMN_INDEX as _,
465            semantic_type: SemanticType::Field,
466            column_schema: ColumnSchema::new(
467                METADATA_SCHEMA_VALUE_COLUMN_NAME,
468                ConcreteDataType::string_datatype(),
469                true,
470            ),
471        };
472
473        // concat region dir
474        let metadata_region_dir = join_dir(&request.region_dir, METADATA_REGION_SUBDIR);
475
476        let options = region_options_for_metadata_region(&request.options);
477        RegionCreateRequest {
478            engine: MITO_ENGINE_NAME.to_string(),
479            column_metadatas: vec![
480                timestamp_column_metadata,
481                key_column_metadata,
482                value_column_metadata,
483            ],
484            primary_key: vec![METADATA_SCHEMA_KEY_COLUMN_INDEX as _],
485            options,
486            region_dir: metadata_region_dir,
487        }
488    }

490    /// Convert [RegionCreateRequest] for data region.
491    ///
492    /// All tag columns in the original request will be converted to value columns.
493    /// Those columns real semantic type is stored in metadata region.
494    ///
495    /// This will also add internal columns to the request.
496    pub fn create_request_for_data_region(
497        &self,
498        request: &RegionCreateRequest,
499    ) -> RegionCreateRequest {
500        let mut data_region_request = request.clone();
501        let mut primary_key = vec![ReservedColumnId::table_id(), ReservedColumnId::tsid()];
502
503        // concat region dir
504        data_region_request.region_dir = join_dir(&request.region_dir, DATA_REGION_SUBDIR);
505
506        // change nullability for tag columns
507        data_region_request
508            .column_metadatas
509            .iter_mut()
510            .for_each(|metadata| {
511                if metadata.semantic_type == SemanticType::Tag {
512                    metadata.column_schema.set_nullable();
513                    primary_key.push(metadata.column_id);
514                }
515            });
516
517        // add internal columns
518        let [table_id_col, tsid_col] = Self::internal_column_metadata();
519        data_region_request.column_metadatas.push(table_id_col);
520        data_region_request.column_metadatas.push(tsid_col);
521        data_region_request.primary_key = primary_key;
522
523        // set data region options
524        set_data_region_options(
525            &mut data_region_request.options,
526            self.config.experimental_sparse_primary_key_encoding,
527        );
528
529        data_region_request
530    }

532    /// Generate internal column metadata.
533    ///
534    /// Return `[table_id_col, tsid_col]`
535    fn internal_column_metadata() -> [ColumnMetadata; 2] {
536        // Safety: BloomFilter is a valid skipping index type
537        let metric_name_col = ColumnMetadata {
538            column_id: ReservedColumnId::table_id(),
539            semantic_type: SemanticType::Tag,
540            column_schema: ColumnSchema::new(
541                DATA_SCHEMA_TABLE_ID_COLUMN_NAME,
542                ConcreteDataType::uint32_datatype(),
543                false,
544            )
545            .with_skipping_options(SkippingIndexOptions {
546                granularity: DEFAULT_TABLE_ID_SKIPPING_INDEX_GRANULARITY,
547                index_type: datatypes::schema::SkippingIndexType::BloomFilter,
548            })
549            .unwrap(),
550        };
551        let tsid_col = ColumnMetadata {
552            column_id: ReservedColumnId::tsid(),
553            semantic_type: SemanticType::Tag,
554            column_schema: ColumnSchema::new(
555                DATA_SCHEMA_TSID_COLUMN_NAME,
556                ConcreteDataType::uint64_datatype(),
557                false,
558            )
559            .with_inverted_index(false),
560        };
561        [metric_name_col, tsid_col]
562    }

565/// Groups the create logical region requests by physical region id.
566fn group_create_logical_region_requests_by_physical_region_id(
567    requests: Vec<(RegionId, RegionCreateRequest)>,
568) -> Result<HashMap<RegionId, Vec<(RegionId, RegionCreateRequest)>>> {
569    let mut result = HashMap::with_capacity(requests.len());
570    for (region_id, request) in requests {
571        let physical_region_id = parse_physical_region_id(&request)?;
572        result
573            .entry(physical_region_id)
574            .or_insert_with(Vec::new)
575            .push((region_id, request));
576    }
577
578    Ok(result)
579}

581/// Parses the physical region id from the request.
582fn parse_physical_region_id(request: &RegionCreateRequest) -> Result<RegionId> {
583    let physical_region_id_raw = request
584        .options
585        .get(LOGICAL_TABLE_METADATA_KEY)
586        .ok_or(MissingRegionOptionSnafu {}.build())?;
587
588    let physical_region_id: RegionId = physical_region_id_raw
589        .parse::<u64>()
590        .with_context(|_| ParseRegionIdSnafu {
591            raw: physical_region_id_raw,
592        })?
593        .into();
594
595    Ok(physical_region_id)
596}

598/// Creates the region options for metadata region in metric engine.
599pub(crate) fn region_options_for_metadata_region(
600    original: &HashMap<String, String>,
601) -> HashMap<String, String> {
602    let mut metadata_region_options = HashMap::new();
603    metadata_region_options.insert(TTL_KEY.to_string(), FOREVER.to_string());
604
605    if let Some(wal_options) = original.get(WAL_OPTIONS_KEY) {
606        metadata_region_options.insert(WAL_OPTIONS_KEY.to_string(), wal_options.to_string());
607    }
608
609    metadata_region_options
610}

742    #[tokio::test]
743    async fn test_create_request_for_physical_regions() {
744        // original request
745        let options: HashMap<_, _> = [
746            ("ttl".to_string(), "60m".to_string()),
747            ("skip_wal".to_string(), "true".to_string()),
748        ]
749        .into_iter()
750        .collect();
751        let request = RegionCreateRequest {
752            engine: METRIC_ENGINE_NAME.to_string(),
753            column_metadatas: vec![
754                ColumnMetadata {
755                    column_id: 0,
756                    semantic_type: SemanticType::Timestamp,
757                    column_schema: ColumnSchema::new(
758                        "timestamp",
759                        ConcreteDataType::timestamp_millisecond_datatype(),
760                        false,
761                    ),
762                },
763                ColumnMetadata {
764                    column_id: 1,
765                    semantic_type: SemanticType::Tag,
766                    column_schema: ColumnSchema::new(
767                        "tag",
768                        ConcreteDataType::string_datatype(),
769                        false,
770                    ),
771                },
772            ],
773            primary_key: vec![0],
774            options,
775            region_dir: "/test_dir".to_string(),
776        };
777
778        // set up
779        let env = TestEnv::new().await;
780        let engine = MetricEngine::try_new(env.mito(), EngineConfig::default()).unwrap();
781        let engine_inner = engine.inner;
782
783        // check create data region request
784        let data_region_request = engine_inner.create_request_for_data_region(&request);
785        assert_eq!(
786            data_region_request.region_dir,
787            "/test_dir/data/".to_string()
788        );
789        assert_eq!(data_region_request.column_metadatas.len(), 4);
790        assert_eq!(
791            data_region_request.primary_key,
792            vec![ReservedColumnId::table_id(), ReservedColumnId::tsid(), 1]
793        );
794        assert!(data_region_request.options.contains_key("ttl"));
795
796        // check create metadata region request
797        let metadata_region_request = engine_inner.create_request_for_metadata_region(&request);
798        assert_eq!(
799            metadata_region_request.region_dir,
800            "/test_dir/metadata/".to_string()
801        );
802        assert_eq!(
803            metadata_region_request.options.get("ttl").unwrap(),
804            "forever"
805        );
806        assert!(!metadata_region_request.options.contains_key("skip_wal"));
807    }
