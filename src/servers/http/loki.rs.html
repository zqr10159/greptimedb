95pub async fn loki_ingest(
96    State(log_state): State<LogState>,
97    Extension(mut ctx): Extension<QueryContext>,
98    TypedHeader(content_type): TypedHeader<ContentType>,
99    LogTableName(table_name): LogTableName,
100    bytes: Bytes,
101) -> Result<HttpResponse> {
102    ctx.set_channel(Channel::Loki);
103    let ctx = Arc::new(ctx);
104    let table_name = table_name.unwrap_or_else(|| LOKI_TABLE_NAME.to_string());
105    let db = ctx.get_db_string();
106    let db_str = db.as_str();
107    let exec_timer = Instant::now();
108
109    // init schemas
110    let mut schemas = LOKI_INIT_SCHEMAS.clone();
111
112    let mut rows = match content_type {
113        x if x == *JSON_CONTENT_TYPE => handle_json_req(bytes, &mut schemas).await,
114        x if x == *PB_CONTENT_TYPE => handle_pb_req(bytes, &mut schemas).await,
115        _ => UnsupportedContentTypeSnafu { content_type }.fail(),
116    }?;
117
118    // fill Null for missing values
119    for row in rows.iter_mut() {
120        row.resize(schemas.len(), GreptimeValue::default());
121    }
122
123    let rows = Rows {
124        rows: rows.into_iter().map(|values| Row { values }).collect(),
125        schema: schemas,
126    };
127    let ins_req = RowInsertRequest {
128        table_name,
129        rows: Some(rows),
130    };
131    let ins_reqs = RowInsertRequests {
132        inserts: vec![ins_req],
133    };
134
135    let handler = log_state.log_handler;
136    let output = handler.insert(ins_reqs, ctx).await;
137
138    if let Ok(Output {
139        data: OutputData::AffectedRows(rows),
140        meta: _,
141    }) = &output
142    {
143        METRIC_LOKI_LOGS_INGESTION_COUNTER
144            .with_label_values(&[db_str])
145            .inc_by(*rows as u64);
146        METRIC_LOKI_LOGS_INGESTION_ELAPSED
147            .with_label_values(&[db_str, METRIC_SUCCESS_VALUE])
148            .observe(exec_timer.elapsed().as_secs_f64());
149    } else {
150        METRIC_LOKI_LOGS_INGESTION_ELAPSED
151            .with_label_values(&[db_str, METRIC_FAILURE_VALUE])
152            .observe(exec_timer.elapsed().as_secs_f64());
153    }
154
155    let response = GreptimedbV1Response::from_output(vec![output])
156        .await
157        .with_execution_time(exec_timer.elapsed().as_millis() as u64);
158    Ok(response)
159}

161async fn handle_json_req(
162    bytes: Bytes,
163    schemas: &mut Vec<ColumnSchema>,
164) -> Result<Vec<Vec<GreptimeValue>>> {
165    let mut column_indexer: HashMap<String, u16> = HashMap::new();
166    column_indexer.insert(GREPTIME_TIMESTAMP.to_string(), 0);
167    column_indexer.insert(LOKI_LINE_COLUMN.to_string(), 1);
168
169    let payload: serde_json::Value =
170        serde_json::from_slice(bytes.as_ref()).context(ParseJsonSnafu)?;
171
172    let streams = payload
173        .get(STREAMS_KEY)
174        .context(InvalidLokiPayloadSnafu {
175            msg: "missing streams",
176        })?
177        .as_array()
178        .context(InvalidLokiPayloadSnafu {
179            msg: "streams is not an array",
180        })?;
181
182    let mut rows = Vec::with_capacity(1000);
183
184    for (stream_index, stream) in streams.iter().enumerate() {
185        // parse lines first
186        // do not use `?` in case there are multiple streams
187        let lines = unwrap_or_warn_continue!(
188            stream.get(LINES_KEY),
189            "missing values on stream {}",
190            stream_index
191        );
192        let lines = unwrap_or_warn_continue!(
193            lines.as_array(),
194            "values is not an array on stream {}",
195            stream_index
196        );
197
198        // get labels
199        let labels = stream
200            .get(LABEL_KEY)
201            .and_then(|label| label.as_object())
202            .map(|l| {
203                l.iter()
204                    .filter_map(|(k, v)| v.as_str().map(|v| (k.clone(), v.to_string())))
205                    .collect::<BTreeMap<String, String>>()
206            });
207
208        // process each line
209        for (line_index, line) in lines.iter().enumerate() {
210            let line = unwrap_or_warn_continue!(
211                line.as_array(),
212                "missing line on stream {} index {}",
213                stream_index,
214                line_index
215            );
216            if line.len() < 2 {
217                warn!(
218                    "line on stream {} index {} is too short",
219                    stream_index, line_index
220                );
221                continue;
222            }
223            // get ts
224            let ts = unwrap_or_warn_continue!(
225                line.first()
226                    .and_then(|ts| ts.as_str())
227                    .and_then(|ts| ts.parse::<i64>().ok()),
228                "missing or invalid timestamp on stream {} index {}",
229                stream_index,
230                line_index
231            );
232            // get line
233            let line_text = unwrap_or_warn_continue!(
234                line.get(1)
235                    .and_then(|line| line.as_str())
236                    .map(|line| line.to_string()),
237                "missing or invalid line on stream {} index {}",
238                stream_index,
239                line_index
240            );
241
242            let structured_metadata = match line.get(2) {
243                Some(sdata) if sdata.is_object() => sdata
244                    .as_object()
245                    .unwrap()
246                    .iter()
247                    .filter_map(|(k, v)| v.as_str().map(|s| (k.clone(), Value::String(s.into()))))
248                    .collect(),
249                _ => BTreeMap::new(),
250            };
251            let structured_metadata = Value::Object(structured_metadata);
252
253            let mut row = init_row(schemas.len(), ts, line_text, structured_metadata);
254
255            process_labels(&mut column_indexer, schemas, &mut row, labels.as_ref());
256
257            rows.push(row);
258        }
259    }
260
261    Ok(rows)
262}

264async fn handle_pb_req(
265    bytes: Bytes,
266    schemas: &mut Vec<ColumnSchema>,
267) -> Result<Vec<Vec<GreptimeValue>>> {
268    let decompressed = prom_store::snappy_decompress(&bytes).unwrap();
269    let req = loki_proto::logproto::PushRequest::decode(&decompressed[..])
270        .context(DecodeOtlpRequestSnafu)?;
271
272    let mut column_indexer: HashMap<String, u16> = HashMap::new();
273    column_indexer.insert(GREPTIME_TIMESTAMP.to_string(), 0);
274    column_indexer.insert(LOKI_LINE_COLUMN.to_string(), 1);
275
276    let cnt = req.streams.iter().map(|s| s.entries.len()).sum::<usize>();
277    let mut rows = Vec::with_capacity(cnt);
278
279    for stream in req.streams {
280        let labels = parse_loki_labels(&stream.labels)
281            .inspect_err(|e| {
282                error!(e; "failed to parse loki labels");
283            })
284            .ok();
285
286        // process entries
287        for entry in stream.entries {
288            let ts = if let Some(ts) = entry.timestamp {
289                ts
290            } else {
291                continue;
292            };
293            let line = entry.line;
294
295            let structured_metadata = entry
296                .structured_metadata
297                .into_iter()
298                .map(|d| (d.name, Value::String(d.value.into())))
299                .collect::<BTreeMap<String, Value>>();
300            let structured_metadata = Value::Object(structured_metadata);
301
302            let mut row = init_row(
303                schemas.len(),
304                prost_ts_to_nano(&ts),
305                line,
306                structured_metadata,
307            );
308
309            process_labels(&mut column_indexer, schemas, &mut row, labels.as_ref());
310
311            rows.push(row);
312        }
313    }
314
315    Ok(rows)
316}

318/// since we're hand-parsing the labels, if any error is encountered, we'll just skip the label
319/// note: pub here for bench usage
320/// ref:
321/// 1. encoding: https://github.com/grafana/alloy/blob/be34410b9e841cc0c37c153f9550d9086a304bca/internal/component/common/loki/client/batch.go#L114-L145
322/// 2. test data: https://github.com/grafana/loki/blob/a24ef7b206e0ca63ee74ca6ecb0a09b745cd2258/pkg/push/types_test.go
323pub fn parse_loki_labels(labels: &str) -> Result<BTreeMap<String, String>> {
324    let mut labels = labels.trim();
325    ensure!(
326        labels.len() >= 2,
327        InvalidLokiLabelsSnafu {
328            msg: "labels string too short"
329        }
330    );
331    ensure!(
332        labels.starts_with("{"),
333        InvalidLokiLabelsSnafu {
334            msg: "missing `{` at the beginning"
335        }
336    );
337    ensure!(
338        labels.ends_with("}"),
339        InvalidLokiLabelsSnafu {
340            msg: "missing `}` at the end"
341        }
342    );
343
344    let mut result = BTreeMap::new();
345    labels = &labels[1..labels.len() - 1];
346
347    while !labels.is_empty() {
348        // parse key
349        let first_index = labels.find("=").with_context(|| InvalidLokiLabelsSnafu {
350            msg: format!("missing `=` near: {}", labels),
351        })?;
352        let key = &labels[..first_index];
353        labels = &labels[first_index + 1..];
354
355        // parse value
356        let qs = quoted_string::parse::<TestSpec>(labels)
357            .map_err(|e| {
358                InvalidLokiLabelsSnafu {
359                    msg: format!(
360                        "failed to parse quoted string near: {}, reason: {}",
361                        labels, e.1
362                    ),
363                }
364                .build()
365            })?
366            .quoted_string;
367
368        labels = &labels[qs.len()..];
369
370        let value = quoted_string::to_content::<TestSpec>(qs).map_err(|e| {
371            InvalidLokiLabelsSnafu {
372                msg: format!("failed to unquote the string: {}, reason: {}", qs, e),
373            }
374            .build()
375        })?;
376
377        // insert key and value
378        result.insert(key.to_string(), value.to_string());
379
380        if labels.is_empty() {
381            break;
382        }
383        ensure!(
384            labels.starts_with(","),
385            InvalidLokiLabelsSnafu { msg: "missing `,`" }
386        );
387        labels = labels[1..].trim_start();
388    }
389
390    Ok(result)
391}

393#[inline]
394fn prost_ts_to_nano(ts: &Timestamp) -> i64 {
395    ts.seconds * 1_000_000_000 + ts.nanos as i64
396}
397
398fn init_row(
399    schema_len: usize,
400    ts: i64,
401    line: String,
402    structured_metadata: Value,
403) -> Vec<GreptimeValue> {
404    // create and init row
405    let mut row = Vec::with_capacity(schema_len);
406    // set ts and line
407    row.push(GreptimeValue {
408        value_data: Some(ValueData::TimestampNanosecondValue(ts)),
409    });
410    row.push(GreptimeValue {
411        value_data: Some(ValueData::StringValue(line)),
412    });
413    row.push(GreptimeValue {
414        value_data: Some(ValueData::BinaryValue(structured_metadata.to_vec())),
415    });
416    for _ in 0..(schema_len - 3) {
417        row.push(GreptimeValue { value_data: None });
418    }
419    row
420}
421
422fn process_labels(
423    column_indexer: &mut HashMap<String, u16>,
424    schemas: &mut Vec<ColumnSchema>,
425    row: &mut Vec<GreptimeValue>,
426    labels: Option<&BTreeMap<String, String>>,
427) {
428    let Some(labels) = labels else {
429        return;
430    };
431
432    // insert labels
433    for (k, v) in labels {
434        if let Some(index) = column_indexer.get(k) {
435            // exist in schema
436            // insert value using index
437            row[*index as usize] = GreptimeValue {
438                value_data: Some(ValueData::StringValue(v.clone())),
439            };
440        } else {
441            // not exist
442            // add schema and append to values
443            schemas.push(ColumnSchema {
444                column_name: k.clone(),
445                datatype: ColumnDataType::String.into(),
446                semantic_type: SemanticType::Tag.into(),
447                datatype_extension: None,
448                options: None,
449            });
450            column_indexer.insert(k.clone(), (schemas.len() - 1) as u16);
451
452            row.push(GreptimeValue {
453                value_data: Some(ValueData::StringValue(v.clone())),
454            });
455        }
456    }
457}

469    fn test_ts_to_nano() {

480    #[test]
481    fn test_parse_loki_labels() {
482        let mut expected = BTreeMap::new();
483        expected.insert("job".to_string(), "foobar".to_string());
484        expected.insert("cluster".to_string(), "foo-central1".to_string());
485        expected.insert("namespace".to_string(), "bar".to_string());
486        expected.insert("container_name".to_string(), "buzz".to_string());
487
488        // perfect case
489        let valid_labels =
490            r#"{job="foobar", cluster="foo-central1", namespace="bar", container_name="buzz"}"#;
491        let re = parse_loki_labels(valid_labels);
492        assert!(re.is_ok());
493        assert_eq!(re.unwrap(), expected);
494
495        // too short
496        let too_short = r#"}"#;
497        let re = parse_loki_labels(too_short);
498        assert!(matches!(re.err().unwrap(), InvalidLokiLabels { .. }));
499
500        // missing start
501        let missing_start = r#"job="foobar"}"#;
502        let re = parse_loki_labels(missing_start);
503        assert!(matches!(re.err().unwrap(), InvalidLokiLabels { .. }));
504
505        // missing start
506        let missing_end = r#"{job="foobar""#;
507        let re = parse_loki_labels(missing_end);
508        assert!(matches!(re.err().unwrap(), InvalidLokiLabels { .. }));
509
510        // missing equal
511        let missing_equal = r#"{job"foobar"}"#;
512        let re = parse_loki_labels(missing_equal);
513        assert!(matches!(re.err().unwrap(), InvalidLokiLabels { .. }));
514
515        // missing quote
516        let missing_quote = r#"{job=foobar}"#;
517        let re = parse_loki_labels(missing_quote);
518        assert!(matches!(re.err().unwrap(), InvalidLokiLabels { .. }));
519
520        // missing comma
521        let missing_comma = r#"{job="foobar" cluster="foo-central1"}"#;
522        let re = parse_loki_labels(missing_comma);
523        assert!(matches!(re.err().unwrap(), InvalidLokiLabels { .. }));
524    }
