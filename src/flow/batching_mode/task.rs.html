86fn determine_query_type(query: &str, query_ctx: &QueryContextRef) -> Result<QueryType, Error> {
87    let stmts =
88        ParserContext::create_with_dialect(query, query_ctx.sql_dialect(), ParseOptions::default())
89            .map_err(BoxedError::new)
90            .context(ExternalSnafu)?;
91
92    ensure!(
93        stmts.len() == 1,
94        InvalidQuerySnafu {
95            reason: format!("Expect only one statement, found {}", stmts.len())
96        }
97    );
98    let stmt = &stmts[0];
99    match stmt {
100        Statement::Tql(_) => Ok(QueryType::Tql),
101        _ => Ok(QueryType::Sql),
102    }
103}

107    /// query is a tql query
108    Tql,
109    /// query is a sql query
110    Sql,

149    /// mark time window range (now - expire_after, now) as dirty (or (0, now) if expire_after not set)
150    ///
151    /// useful for flush_flow to flush dirty time windows range
152    pub fn mark_all_windows_as_dirty(&self) -> Result<(), Error> {
153        let now = SystemTime::now();
154        let now = Timestamp::new_second(
155            now.duration_since(UNIX_EPOCH)
156                .expect("Time went backwards")
157                .as_secs() as _,
158        );
159        let lower_bound = self
160            .config
161            .expire_after
162            .map(|e| now.sub_duration(Duration::from_secs(e as _)))
163            .transpose()
164            .map_err(BoxedError::new)
165            .context(ExternalSnafu)?
166            .unwrap_or(Timestamp::new_second(0));
167        debug!(
168            "Flow {} mark range ({:?}, {:?}) as dirty",
169            self.config.flow_id, lower_bound, now
170        );
171        self.state
172            .write()
173            .unwrap()
174            .dirty_time_windows
175            .add_window(lower_bound, Some(now));
176        Ok(())
177    }

181        &self,
182        engine: &QueryEngineRef,
183        frontend_client: &Arc<FrontendClient>,
184    ) -> Result<Option<(u32, Duration)>, Error> {
185        if !self.is_table_exist(&self.config.sink_table_name).await? {
186            let create_table = self.gen_create_table_expr(engine.clone()).await?;
187            info!(
188                "Try creating sink table(if not exists) with expr: {:?}",
189                create_table
190            );
191            self.create_table(frontend_client, create_table).await?;
192            info!(
193                "Sink table {}(if not exists) created",
194                self.config.sink_table_name.join(".")
195            );
196        }
197
198        Ok(None)
199    }

211        &self,
212        engine: &QueryEngineRef,
213        frontend_client: &Arc<FrontendClient>,
214    ) -> Result<Option<(u32, Duration)>, Error> {

224    pub async fn gen_insert_plan(
225        &self,
226        engine: &QueryEngineRef,
227    ) -> Result<Option<LogicalPlan>, Error> {
228        let (table, df_schema) = get_table_info_df_schema(
229            self.config.catalog_manager.clone(),
230            self.config.sink_table_name.clone(),
231        )
232        .await?;
233
234        let new_query = self
235            .gen_query_with_time_window(engine.clone(), &table.meta.schema)
236            .await?;
237
238        let insert_into = if let Some((new_query, _column_cnt)) = new_query {
239            // first check if all columns in input query exists in sink table
240            // since insert into ref to names in record batch generate by given query
241            let table_columns = df_schema
242                .columns()
243                .into_iter()
244                .map(|c| c.name)
245                .collect::<BTreeSet<_>>();
246            for column in new_query.schema().columns() {
247                ensure!(
248                    table_columns.contains(column.name()),
249                    InvalidQuerySnafu {
250                        reason: format!(
251                            "Column {} not found in sink table with columns {:?}",
252                            column, table_columns
253                        ),
254                    }
255                );
256            }
257            // update_at& time index placeholder (if exists) should have default value
258            LogicalPlan::Dml(DmlStatement::new(
259                datafusion_common::TableReference::Full {
260                    catalog: self.config.sink_table_name[0].clone().into(),
261                    schema: self.config.sink_table_name[1].clone().into(),
262                    table: self.config.sink_table_name[2].clone().into(),
263                },
264                df_schema,
265                WriteOp::Insert(datafusion_expr::dml::InsertOp::Append),
266                Arc::new(new_query),
267            ))
268        } else {
269            return Ok(None);
270        };
271        let insert_into = insert_into.recompute_schema().context(DatafusionSnafu {
272            context: "Failed to recompute schema",
273        })?;
274        Ok(Some(insert_into))
275    }

290    pub async fn execute_logical_plan(
291        &self,
292        frontend_client: &Arc<FrontendClient>,
293        plan: &LogicalPlan,
294    ) -> Result<Option<(u32, Duration)>, Error> {
295        let instant = Instant::now();
296        let flow_id = self.config.flow_id;
297
298        debug!(
299            "Executing flow {flow_id}(expire_after={:?} secs) with query {}",
300            self.config.expire_after, &plan
301        );
302
303        let catalog = &self.config.sink_table_name[0];
304        let schema = &self.config.sink_table_name[1];
305
306        // fix all table ref by make it fully qualified, i.e. "table_name" => "catalog_name.schema_name.table_name"
307        let fixed_plan = plan
308            .clone()
309            .transform_down_with_subqueries(|p| {
310                if let LogicalPlan::TableScan(mut table_scan) = p {
311                    let resolved = table_scan.table_name.resolve(catalog, schema);
312                    table_scan.table_name = resolved.into();
313                    Ok(Transformed::yes(LogicalPlan::TableScan(table_scan)))
314                } else {
315                    Ok(Transformed::no(p))
316                }
317            })
318            .with_context(|_| DatafusionSnafu {
319                context: format!("Failed to fix table ref in logical plan, plan={:?}", plan),
320            })?
321            .data;
322
323        let expanded_plan = CountWildcardRule::new()
324            .analyze(fixed_plan.clone(), &Default::default())
325            .with_context(|_| DatafusionSnafu {
326                context: format!(
327                    "Failed to expand wildcard in logical plan, plan={:?}",
328                    fixed_plan
329                ),
330            })?;
331
332        let plan = expanded_plan;
333        let mut peer_desc = None;
334
335        let res = {
336            let _timer = METRIC_FLOW_BATCHING_ENGINE_QUERY_TIME
337                .with_label_values(&[flow_id.to_string().as_str()])
338                .start_timer();
339
340            // hack and special handling the insert logical plan
341            let req = if let Some((insert_to, insert_plan)) =
342                breakup_insert_plan(&plan, catalog, schema)
343            {
344                let message = DFLogicalSubstraitConvertor {}
345                    .encode(&insert_plan, DefaultSerializer)
346                    .context(SubstraitEncodeLogicalPlanSnafu)?;
347                api::v1::greptime_request::Request::Query(api::v1::QueryRequest {
348                    query: Some(api::v1::query_request::Query::InsertIntoPlan(
349                        api::v1::InsertIntoPlan {
350                            table_name: Some(insert_to),
351                            logical_plan: message.to_vec(),
352                        },
353                    )),
354                })
355            } else {
356                let message = DFLogicalSubstraitConvertor {}
357                    .encode(&plan, DefaultSerializer)
358                    .context(SubstraitEncodeLogicalPlanSnafu)?;
359
360                api::v1::greptime_request::Request::Query(api::v1::QueryRequest {
361                    query: Some(api::v1::query_request::Query::LogicalPlan(message.to_vec())),
362                })
363            };
364
365            frontend_client
366                .handle(req, catalog, schema, &mut peer_desc)
367                .await
368        };
369
370        let elapsed = instant.elapsed();
371        if let Ok(affected_rows) = &res {
372            debug!(
373                "Flow {flow_id} executed, affected_rows: {affected_rows:?}, elapsed: {:?}",
374                elapsed
375            );
376            METRIC_FLOW_ROWS
377                .with_label_values(&[format!("{}-out-batching", flow_id).as_str()])
378                .inc_by(*affected_rows as _);
379        } else if let Err(err) = &res {
380            warn!(
381                "Failed to execute Flow {flow_id} on frontend {:?}, result: {err:?}, elapsed: {:?} with query: {}",
382                peer_desc, elapsed, &plan
383            );
384        }
385
386        // record slow query
387        if elapsed >= SLOW_QUERY_THRESHOLD {
388            warn!(
389                "Flow {flow_id} on frontend {:?} executed for {:?} before complete, query: {}",
390                peer_desc, elapsed, &plan
391            );
392            METRIC_FLOW_BATCHING_ENGINE_SLOW_QUERY
393                .with_label_values(&[
394                    flow_id.to_string().as_str(),
395                    &peer_desc.unwrap_or_default().to_string(),
396                ])
397                .observe(elapsed.as_secs_f64());
398        }
399
400        self.state
401            .write()
402            .unwrap()
403            .after_query_exec(elapsed, res.is_ok());
404
405        let res = res?;
406
407        Ok(Some((res, elapsed)))
408    }

410    /// start executing query in a loop, break when receive shutdown signal
411    ///
412    /// any error will be logged when executing query
413    pub async fn start_executing_loop(
414        &self,
415        engine: QueryEngineRef,
416        frontend_client: Arc<FrontendClient>,
417    ) {
418        let flow_id_str = self.config.flow_id.to_string();
419        loop {
420            // first check if shutdown signal is received
421            // if so, break the loop
422            {
423                let mut state = self.state.write().unwrap();
424                match state.shutdown_rx.try_recv() {
425                    Ok(()) => break,
426                    Err(TryRecvError::Closed) => {
427                        warn!(
428                            "Unexpected shutdown flow {}, shutdown anyway",
429                            self.config.flow_id
430                        );
431                        break;
432                    }
433                    Err(TryRecvError::Empty) => (),
434                }
435            }
436            METRIC_FLOW_BATCHING_ENGINE_START_QUERY_CNT
437                .with_label_values(&[&flow_id_str])
438                .inc();
439
440            let new_query = match self.gen_insert_plan(&engine).await {
441                Ok(new_query) => new_query,
442                Err(err) => {
443                    common_telemetry::error!(err; "Failed to generate query for flow={}", self.config.flow_id);
444                    // also sleep for a little while before try again to prevent flooding logs
445                    tokio::time::sleep(MIN_REFRESH_DURATION).await;
446                    continue;
447                }
448            };
449
450            let res = if let Some(new_query) = &new_query {
451                self.execute_logical_plan(&frontend_client, new_query).await
452            } else {
453                Ok(None)
454            };
455
456            match res {
457                // normal execute, sleep for some time before doing next query
458                Ok(Some(_)) => {
459                    let sleep_until = {
460                        let state = self.state.write().unwrap();
461
462                        state.get_next_start_query_time(
463                            self.config.flow_id,
464                            &self
465                                .config
466                                .time_window_expr
467                                .as_ref()
468                                .and_then(|t| *t.time_window_size()),
469                            Some(DEFAULT_BATCHING_ENGINE_QUERY_TIMEOUT),
470                        )
471                    };
472                    tokio::time::sleep_until(sleep_until).await;
473                }
474                // no new data, sleep for some time before checking for new data
475                Ok(None) => {
476                    debug!(
477                        "Flow id = {:?} found no new data, sleep for {:?} then continue",
478                        self.config.flow_id, MIN_REFRESH_DURATION
479                    );
480                    tokio::time::sleep(MIN_REFRESH_DURATION).await;
481                    continue;
482                }
483                // TODO(discord9): this error should have better place to go, but for now just print error, also more context is needed
484                Err(err) => {
485                    METRIC_FLOW_BATCHING_ENGINE_ERROR_CNT
486                        .with_label_values(&[&flow_id_str])
487                        .inc();
488                    match new_query {
489                        Some(query) => {
490                            common_telemetry::error!(err; "Failed to execute query for flow={} with query: {query}", self.config.flow_id)
491                        }
492                        None => {
493                            common_telemetry::error!(err; "Failed to generate query for flow={}", self.config.flow_id)
494                        }
495                    }
496                    // also sleep for a little while before try again to prevent flooding logs
497                    tokio::time::sleep(MIN_REFRESH_DURATION).await;
498                }
499            }
500        }
501    }

519    /// will merge and use the first ten time window in query
520    async fn gen_query_with_time_window(
521        &self,
522        engine: QueryEngineRef,
523        sink_table_schema: &Arc<Schema>,
524    ) -> Result<Option<(LogicalPlan, usize)>, Error> {
525        let query_ctx = self.state.read().unwrap().query_ctx.clone();
526        let start = SystemTime::now();
527        let since_the_epoch = start
528            .duration_since(UNIX_EPOCH)
529            .expect("Time went backwards");
530        let low_bound = self
531            .config
532            .expire_after
533            .map(|e| since_the_epoch.as_secs() - e as u64)
534            .unwrap_or(u64::MIN);
535
536        let low_bound = Timestamp::new_second(low_bound as i64);
537        let schema_len = self.config.output_schema.fields().len();
538
539        let expire_time_window_bound = self
540            .config
541            .time_window_expr
542            .as_ref()
543            .map(|expr| expr.eval(low_bound))
544            .transpose()?;
545
546        let (Some((Some(l), Some(u))), QueryType::Sql) =
547            (expire_time_window_bound, &self.config.query_type)
548        else {
549            // either no time window or not a sql query, then just use the original query
550            // use sink_table_meta to add to query the `update_at` and `__ts_placeholder` column's value too for compatibility reason
551            debug!(
552                "Flow id = {:?}, can't get window size: precise_lower_bound={expire_time_window_bound:?}, using the same query", self.config.flow_id
553            );
554            // clean dirty time window too, this could be from create flow's check_execute
555            self.state.write().unwrap().dirty_time_windows.clean();
556
557            // TODO(discord9): not add auto column for tql query?
558            let mut add_auto_column = AddAutoColumnRewriter::new(sink_table_schema.clone());
559
560            let plan = sql_to_df_plan(query_ctx.clone(), engine.clone(), &self.config.query, false)
561                .await?;
562
563            let plan = plan
564                .clone()
565                .rewrite(&mut add_auto_column)
566                .with_context(|_| DatafusionSnafu {
567                    context: format!("Failed to rewrite plan:\n {}\n", plan),
568                })?
569                .data;
570            let schema_len = plan.schema().fields().len();
571
572            // since no time window lower/upper bound is found, just return the original query(with auto columns)
573            return Ok(Some((plan, schema_len)));
574        };
575
576        debug!(
577            "Flow id = {:?}, found time window: precise_lower_bound={:?}, precise_upper_bound={:?}",
578            self.config.flow_id, l, u
579        );
580        let window_size = u.sub(&l).with_context(|| UnexpectedSnafu {
581            reason: format!("Can't get window size from {u:?} - {l:?}"),
582        })?;
583        let col_name = self
584            .config
585            .time_window_expr
586            .as_ref()
587            .map(|expr| expr.column_name.clone())
588            .with_context(|| UnexpectedSnafu {
589                reason: format!(
590                    "Flow id={:?}, Failed to get column name from time window expr",
591                    self.config.flow_id
592                ),
593            })?;
594
595        let expr = self
596            .state
597            .write()
598            .unwrap()
599            .dirty_time_windows
600            .gen_filter_exprs(
601                &col_name,
602                Some(l),
603                window_size,
604                DirtyTimeWindows::MAX_FILTER_NUM,
605                self.config.flow_id,
606                Some(self),
607            )?;
608
609        debug!(
610            "Flow id={:?}, Generated filter expr: {:?}",
611            self.config.flow_id,
612            expr.as_ref()
613                .map(|expr| expr_to_sql(expr).with_context(|_| DatafusionSnafu {
614                    context: format!("Failed to generate filter expr from {expr:?}"),
615                }))
616                .transpose()?
617                .map(|s| s.to_string())
618        );
619
620        let Some(expr) = expr else {
621            // no new data, hence no need to update
622            debug!("Flow id={:?}, no new data, not update", self.config.flow_id);
623            return Ok(None);
624        };
625
626        let mut add_filter = AddFilterRewriter::new(expr);
627        let mut add_auto_column = AddAutoColumnRewriter::new(sink_table_schema.clone());
628
629        let plan =
630            sql_to_df_plan(query_ctx.clone(), engine.clone(), &self.config.query, false).await?;
631        let rewrite = plan
632            .clone()
633            .rewrite(&mut add_filter)
634            .and_then(|p| p.data.rewrite(&mut add_auto_column))
635            .with_context(|_| DatafusionSnafu {
636                context: format!("Failed to rewrite plan:\n {}\n", plan),
637            })?
638            .data;
639        // only apply optimize after complex rewrite is done
640        let new_plan = apply_df_optimizer(rewrite).await?;
641
642        Ok(Some((new_plan, schema_len)))
643    }
644}

646// auto created table have a auto added column `update_at`, and optional have a `AUTO_CREATED_PLACEHOLDER_TS_COL` column for time index placeholder if no timestamp column is specified
647// TODO(discord9): for now no default value is set for auto added column for compatibility reason with streaming mode, but this might change in favor of simpler code?
648fn create_table_with_expr(
649    plan: &LogicalPlan,
650    sink_table_name: &[String; 3],
651) -> Result<CreateTableExpr, Error> {
652    let fields = plan.schema().fields();
653    let (first_time_stamp, primary_keys) = build_primary_key_constraint(plan, fields)?;
654
655    let mut column_schemas = Vec::new();
656    for field in fields {
657        let name = field.name();
658        let ty = ConcreteDataType::from_arrow_type(field.data_type());
659        let col_schema = if first_time_stamp == Some(name.clone()) {
660            ColumnSchema::new(name, ty, false).with_time_index(true)
661        } else {
662            ColumnSchema::new(name, ty, true)
663        };
664        column_schemas.push(col_schema);
665    }
666
667    let update_at_schema = ColumnSchema::new(
668        AUTO_CREATED_UPDATE_AT_TS_COL,
669        ConcreteDataType::timestamp_millisecond_datatype(),
670        true,
671    );
672    column_schemas.push(update_at_schema);
673
674    let time_index = if let Some(time_index) = first_time_stamp {
675        time_index
676    } else {
677        column_schemas.push(
678            ColumnSchema::new(
679                AUTO_CREATED_PLACEHOLDER_TS_COL,
680                ConcreteDataType::timestamp_millisecond_datatype(),
681                false,
682            )
683            .with_time_index(true),
684        );
685        AUTO_CREATED_PLACEHOLDER_TS_COL.to_string()
686    };
687
688    let column_defs =
689        column_schemas_to_defs(column_schemas, &primary_keys).context(ConvertColumnSchemaSnafu)?;
690    Ok(CreateTableExpr {
691        catalog_name: sink_table_name[0].clone(),
692        schema_name: sink_table_name[1].clone(),
693        table_name: sink_table_name[2].clone(),
694        desc: "Auto created table by flow engine".to_string(),
695        column_defs,
696        time_index,
697        primary_keys,
698        create_if_not_exists: true,
699        table_options: Default::default(),
700        table_id: None,
701        engine: "mito".to_string(),
702    })
703}

705/// Return first timestamp column which is in group by clause and other columns which are also in group by clause
706///
707/// # Returns
708///
709/// * `Option<String>` - first timestamp column which is in group by clause
710/// * `Vec<String>` - other columns which are also in group by clause
711fn build_primary_key_constraint(
712    plan: &LogicalPlan,
713    schema: &Fields,
714) -> Result<(Option<String>, Vec<String>), Error> {
715    let mut pk_names = FindGroupByFinalName::default();
716
717    plan.visit(&mut pk_names)
718        .with_context(|_| DatafusionSnafu {
719            context: format!("Can't find aggr expr in plan {plan:?}"),
720        })?;
721
722    // if no group by clause, return empty
723    let pk_final_names = pk_names.get_group_expr_names().unwrap_or_default();
724    if pk_final_names.is_empty() {
725        return Ok((None, Vec::new()));
726    }
727
728    let all_pk_cols: Vec<_> = schema
729        .iter()
730        .filter(|f| pk_final_names.contains(f.name()))
731        .map(|f| f.name().clone())
732        .collect();
733    // auto create table use first timestamp column in group by clause as time index
734    let first_time_stamp = schema
735        .iter()
736        .find(|f| {
737            all_pk_cols.contains(&f.name().clone())
738                && ConcreteDataType::from_arrow_type(f.data_type()).is_timestamp()
739        })
740        .map(|f| f.name().clone());
741
742    let all_pk_cols: Vec<_> = all_pk_cols
743        .into_iter()
744        .filter(|col| first_time_stamp != Some(col.to_string()))
745        .collect();
746
747    Ok((first_time_stamp, all_pk_cols))
748}
